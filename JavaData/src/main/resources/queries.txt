SELECT 
	stops.stop_id,
	stops.stop_lat, 	 
	stops.stop_lon, 
	stops.stop_name 
FROM routes
JOIN trips ON trips.route_id = routes.route_id
JOIN stop_times ON stop_times.trip_id = trips.trip_id
JOIN stops ON stops.stop_id = stop_times.stop_id
WHERE routes.route_type = 2
GROUP BY stop_name;
